
# This code was used to make the surface area growth lineplot in Supplementary Figure S4
# Load required packages

library(ggplot2)
library(dplyr)
library(readxl)


# Set the file path
file_path <- "C:\\Users\\askin\\OneDrive\\Documents\\Adrienne_Lab_Postdoc_Data\\vos3_paper_writing\\Supplementary codes and tables\\Data for plots\\Figure S4\\WT_and_as2-6D_surface_area.xlsx"

# Import data from Excel
df <- read_excel(file_path,sheet = 1)
df$Timepoint <- factor(df$Timepoint, levels = c("0_hr", "24_hr", "48_hr"))

# Calculate summary statistics
df_summary <- df %>%
  group_by(Genotype, Timepoint, Side) %>%
  summarise(
    Mean = mean(Surface_Area),
    SEM = sd(Surface_Area) / sqrt(n()),
    .groups = "drop"
  )

# Create the plot
plot <- ggplot(df_summary, aes(x = Timepoint, y = Mean, group = interaction(Genotype, Side))) +
  geom_errorbar(aes(ymin = Mean - SEM, ymax = Mean + SEM, color = Genotype), width = 0.1) +
  geom_point(aes(color = Genotype), size = 3) +
  geom_line(aes(linetype = Side, color = Genotype), size = 1) +
  scale_color_manual(values = c("WT" = "black", "as2-6D" = "red")) +
  scale_linetype_manual(values = c("dashed", "solid")) +
  labs(x = "Timepoint", y = "Average Surface Area") +
  scale_y_continuous(breaks = seq(0, max(df_summary$Mean + df_summary$SEM), by = 5000)) +
  theme_classic() +
  theme(legend.title = element_blank(),  axis.line = element_line(size = 0.5, colour = "black"))

# Save the plot as a high-quality PNG image
ggsave("WT and as2-6D_linePlot.png", plot, width = 10, height = 6, dpi = 600)

# Display the plot in the R console
print(plot)
